Research question: Do the number of measles cases change depending on the time of year?

monthlyStats <- function(monthly_measles) {
  tibble(mean_measles = mean(monthly_measles, na.rm = TRUE),
         median_measles = median(monthly_measles, na.rm = TRUE),
         max_measles = max(monthly_measles, na.rm = TRUE))
}

measles_by_month <- tibble()

for (i in 1:12) {
  monthly_measles <- cases_month |>
    filter(month == i) |> 
    pull(measles_total)
  
  measles_by_month <- bind_rows(measles_by_month, 
                                monthlyStats(monthly_measles) |>
                                  mutate(month = i))
}
measles_by_month <- measles_by_month |>
  select(month, mean_measles, median_measles, max_measles)

Conclusion

According to our resulting summaries, it appears the number of measles cases tends to be the most during the beginning of the year (January - March), while the number of cases seems to be the lowest during the ansition between summer and fall (August - October). However, it’s important to keep in mind that these are summaries over the last 9 years, and include cases all over the world (and seasons/environment are different depending on location).

Research question: Has the total number of reported measles cases increased or decreased over time globally?

trend_label <- function(current, previous) {
  pct_change <- (current - previous) / previous * 100
  case_when(
    pct_change < -20 ~ "declining",
    pct_change >  20 ~ "surging",
    .default = "stable"
  )
}

yearly_summary <- cases_year |>
  group_by(year) |>
  summarise(
    total_cases = sum(measles_total, na.rm = TRUE),
    n_countries = n_distinct(country),
    .groups = "drop"
  ) |>
  arrange(year) |>
  mutate(
    prev_total = lag(total_cases),
    pct_change = round((total_cases - prev_total) / prev_total * 100, 1),
    trend = trend_label(total_cases, prev_total)
  ) |>
  select(year, total_cases, n_countries, pct_change, trend)

print(yearly_summary)

Using Existing Columns + Iteration for Summary Statistics

max <- nrow(cases_year)

for (i in 1:nrow(cases_year)) {
  lab <- cases_year$measles_lab_confirmed[i]
  epi <- cases_year$measles_epi_linked[i]
  clinic <- cases_year$measles_clinical[i]

  if (lab >= epi & lab >= clinic){
    max[i] <- "lab"
    
  } else if (epi >= lab & epi >= clinic) {
    max[i] <- "epi"
    
  } else {
    max[i] <- "clinic"
  }
  
}

max_tbl <- tibble(method = max)

sum_prop <- max_tbl |> 
  group_by(method)|> 
  summarise(count = n(), prop = n() / nrow(max_tbl))

sum_prop

Research Question: Which country and region have the most measles and rubella cases combined?

total_country_cases <- cases_year |>
  mutate(
    across(c(measles_total, rubella_total), as.numeric)) |> # convert columns to numbers
  group_by(region, country) |> # group by region and country
  summarize(
    total_measles = sum(measles_total),
    total_rubella = sum(rubella_total),
    .groups = "drop") |>
  mutate(
    total_cases = total_measles + total_rubella) |> # new column, measles and rubella cases combined
arrange(desc(total_cases)) # sorts from highest to lowest cases
