Writing function

Next, using the methods we used for Canada as a blueprint, I’ll create a function to get the data for all countries.

get_gdp_data <- function(country, year, key = my_key) {
  if (is.na(country) || is.na(year)) {
    return(NULL)
  }

  country <- str_replace_all(country, " ", "%20")
  base_url <- "https://api.api-ninjas.com/v1/gdp"
  url <- str_c(base_url, "?country=", country, "&year=", year)
  raw_data <- GET(url, add_headers("X-Api-Key" = my_key))
  json_data <- rawToChar(raw_data$content)
  tidy_data <- fromJSON(json_data)
  if (!is.data.frame(tidy_data) || nrow(tidy_data) == 0) {
    return(NULL)
  }
  if (!"gdp_per_capita_nominal" %in% names(tidy_data) || 
      is.na(tidy_data$gdp_per_capita_nominal)) {
    return(tibble(
      country = country,
      year = year,
      gdp_per_capita_nominal = NA
    ))
  }
  gdps <- tidy_data %>% select(country, year, gdp_per_capita_nominal)
}

Brief description of new dataset

We now have a new dataframe (which I’ve named “results”) which contains 2380 rows with 3 variables. As the name suggests, this column contains the nominal GDP per capita for that country, in that year. This was taken from API Ninja’s dataset.

Creating a function to extract total_aqi for each country

get_aqi <- function(city, country){
  
  base_url <- "https://api.api-ninjas.com/v1/airquality?"
  
  city <- paste0("city=", gsub(" ", "%20", city))
  
  country <- paste0("country=", gsub(" ", "%20", country))

  my_url <- glue("{base_url}{city}&{country}")

  my_key = "YOUR_API_KEY_HERE"
  
  my_row <- GET(my_url, add_headers("X-Api-Key" = my_key))

  my_data <- fromJSON(rawToChar(my_row$content))
  
  if (length(my_data$overall_aqi) == 0){
    
    my_total_aqi <- NA
    
  }else {
    
    my_total_aqi <- my_data$overall_aqi
    
  }
  
  return(my_total_aqi)
  
}

Research Question: How does the relationship between GDP per capita and measles incidence rate change before and after COVID?

meta_df |>
  mutate(
    time_period = case_when(
      year >= 2015 & year <= 2019 ~ "Pre-COVID (2015-2019)",
      year >= 2020 & year <= 2023 ~ "Post-COVID (2020-2023)"),
      time_period = factor(time_period, levels = c("Pre-COVID (2015-2019)", "Post-COVID (2020-2023)"))) |>
  filter(!is.na(time_period)) |>
  group_by(iso3, country_full, region, time_period) |>

    avg_incidence = mean(measles_incidence_rate_per_1000000_total_population, na.rm = TRUE),
