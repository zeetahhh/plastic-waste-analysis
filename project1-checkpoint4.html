### Visualization: Average AQI vs. Average Measles Cases by Country (2020–2025)
meta_df %>%
  filter(!is.na(total_aqi), !is.na(total_suspected_measles_rubella_cases)) %>%
  group_by(country, total_aqi) %>%
  summarise(avg_cases = mean(total_suspected_measles_rubella_cases, na.rm = TRUE)) %>%
  ggplot(aes(x = total_aqi, y = avg_cases)) +
  geom_point(alpha = 0.6, color = "dodgerblue") +
  geom_smooth(method = "lm", se = TRUE, color = "red") +
  scale_y_log10(labels = scales::comma) +
  labs(
    title = "Air Quality Index vs. Average Measles Cases by Country",
    subtitle = "Each point represents one country; cases shown on log scale",
    x = "Air Quality Index (higher = worse air quality)",
    y = "Average Annual Measles Cases (log scale)",
    caption = "Sources: API Ninjas AQI data; TidyTuesday measles dataset"
  )
aqi_model <- meta_df |>
  filter(!is.na(total_aqi), !is.na(total_suspected_measles_rubella_cases)) |>
  group_by(country, total_aqi) |>
  summarise(avg_cases = mean(total_suspected_measles_rubella_cases, na.rm = TRUE),
            .groups = "drop") |>
  mutate(log_avg_cases = log10(avg_cases + 1))

# Fitting the model
model <- lm(log_avg_cases ~ total_aqi, data = aqi_model)
summary(model)

Conclusion
